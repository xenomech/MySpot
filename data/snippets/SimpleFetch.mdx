Fetch random thumbnail images from <a href="https://jsonplaceholder.typicode.com/">jsonplaceholder</a> on page load and display it.
(using axios for CRUD operations)

<a
  href="https://codesandbox.io/s/fetch-random-data-l190v?file=/src/App.js"
  target="_blank"
>
  Visit codeSandbox
</a>

```js
import axios from "axios";
import { useEffect, useState } from "react";

export default function App() {
  const [imgdat, setImgdat] = useState([]);
  useEffect(() => {
    axios
      .get("https://jsonplaceholder.typicode.com/photos")
      .then(({ data }) => {
        setImgdat(data);
      });
  }, []);

  return (
    <div className="App">
      {imgdat.map((item, index) => {
        return <img alt="data" src={item.thumbnailUrl} key={index} />;
      })}
      <h1>Data from JSON Placeholder</h1>
    </div>
  );
}
```
